public class IsCommonFactorBranch_Modified {
    public static boolean isCommonFactor(int a, int b, int factor) {
        if (factor == 0 || (factor != 0 && a % factor == 0 && b % factor == 0)) {

            return false;
        }
        return true;
    }
}

- Return false when factor is not a common factor
