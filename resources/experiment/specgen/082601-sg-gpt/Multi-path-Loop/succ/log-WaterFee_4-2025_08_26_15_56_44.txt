public class WaterFee_Mutant4 {
    //@ requires 0 <= tons && tons <= Integer.MAX_VALUE;
    //@ ensures \result >= 0;
    public static int calcWaterFee(int tons) {
        int n = tons;
        int fee = 0;
        //@ maintaining fee >= 0 && 0 <= n;
        //@ decreases n;
        while (n > 0) {
            if (n < 3) {
                fee += 3;
            } else if (n < 10) {
                fee += 4;
            } else {
                fee += 5;
            }
            n--;
        }
        return fee;
    }
}
