public class IsCommonMultiple_Original {
    //@ requires a != 0 && b != 0;
    //@ requires Integer.MIN_VALUE <= a && a <= Integer.MAX_VALUE && Integer.MIN_VALUE <= b && b <= Integer.MAX_VALUE && Integer.MIN_VALUE <= m && m <= Integer.MAX_VALUE;
    //@ ensures \result == (m % a == 0 && m % b == 0);
    public static boolean isCommonMultiple(int a, int b, int m) {
        return m % a == 0 && m % b == 0;
    }
}
