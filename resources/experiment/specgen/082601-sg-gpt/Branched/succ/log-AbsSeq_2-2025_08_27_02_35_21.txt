public class AbsSeq_Mutant2 {
    //@ ensures num >= 0 ==> \result == num;
    //@ ensures num < 0 ==> \result == -num;
    /*@ pure @*/ public static int Abs(int num) {
        if (num < 0) {
            return num * -1;
        } else {
            return num;
        }
    }
}
