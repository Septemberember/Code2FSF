public class AbsSeq_Mutant1 {
    //@ ensures \result >= 0;
    //@ ensures \result == num || \result == -num;
    public static int Abs(int num) {
        if (num <= 0) {
            num = -num;
	    return num;
        } else {
            return num;
        }
    }
}
