public class SmallestEvenMulBranch_Original {

    //@ requires n >= 0;
    //@ ensures \result % 2 == 0;
    //@ ensures \result >= n;
    public static int smallestEvenMultiple(int n) {
        if (n % 2 == 0) {
            return n;
        }
        return 2 * n;
    }
}
