public class SafeLanding_Mutant1 {
    //@ requires Integer.MIN_VALUE <= height && height <= Integer.MAX_VALUE;
    //@ requires Integer.MIN_VALUE <= speed && speed <= Integer.MAX_VALUE;
    //@ requires Integer.MIN_VALUE <= tilt && tilt <= Integer.MAX_VALUE;
    //@ ensures \result == 1 || \result == 0;
    public static int safeLanding(int height, int speed, int tilt) {
        int safe = 1;
        if (height <= 5) {
            safe = 0;
        }
        if (speed > 3) {
            safe = 0;
        }
        if (tilt > 10) {
            safe = 0;
        }
        return safe;
    }
}
