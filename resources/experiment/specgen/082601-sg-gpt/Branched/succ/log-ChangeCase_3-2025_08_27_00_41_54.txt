public class ChangeCase_Mutant3 {

    //@ ensures \result == c || (\result >= 'A' && \result <= 'Z') || (\result >= 'a' && \result <= 'z');
    public static char changeCase(char c) {
        char result = ' ';
        if (c > 'z') {
            result = c;
        } else if (c >= 'a') {
            result = (char) (c - 'a' + 'A');
        } else if (c > 'Z') {
            result = c;
        } else if (c >= 'A') {
            result = (char) (c - 'A' + 'a');
        } else {
            result = c;
        }
        return result;
    }
}
