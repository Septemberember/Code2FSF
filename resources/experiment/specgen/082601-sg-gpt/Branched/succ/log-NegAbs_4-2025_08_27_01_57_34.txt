public class NegAbs_Mutant4 {
    //@ requires num >= Integer.MIN_VALUE && num <= Integer.MAX_VALUE;
    //@ ensures \result >= 0;
    //@ ensures \result == Math.abs(num);
    public static int negAbs(int num) {
        if (num > 0)
            return num;
        else
            return -num;
    }
}
