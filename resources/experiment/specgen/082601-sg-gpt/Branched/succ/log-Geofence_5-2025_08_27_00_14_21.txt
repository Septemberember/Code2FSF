public class Geofence_Mutant5 {
    //@ ensures \result == 2 || \result == 1;
    public static int geofence(int x, int y, int limitX, int limitY) {
        int alarm = 2;
        if (x < 0 || x > limitX) {
            alarm = 1;
        }
        if (y < 0 || y > limitY) {
            alarm = 1;
        }
        return alarm;
    }
}
