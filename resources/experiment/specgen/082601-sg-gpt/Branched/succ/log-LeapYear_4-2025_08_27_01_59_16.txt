public class LeapYear_Mutant4 {
    //@ requires Integer.MIN_VALUE <= year && year <= Integer.MAX_VALUE;
    //@ ensures \result ==> (year % 4 == 0 && (year % 100 != 0 || year % 400 == 0));
    //@ ensures !\result ==> !(year % 4 == 0 && (year % 100 != 0 || year % 400 == 0));
    public static boolean isLeapYear(int year) {
        boolean leap = false;
        if (year % 4 == 0) {
            if (year % 100 == 0) {
                if (year % 400 == 0)
                    leap = true;
            } else
                leap = true;
        } else
            leap = false;
        return leap;
    }
}
