public class GPSCheck_Mutant5 {
    //@ requires true;
    //@ ensures \result == 2 <==> satellites < 4 && signalStrength < 20;
    //@ ensures \result == 0 <==> !(satellites < 4 && signalStrength < 20);
    public static int GPSCheck(int satellites, int signalStrength) {
        int alarm = 0;
        if (satellites < 4 && signalStrength < 20) {
            alarm = 2;
        }
        return alarm;
    }
}
