public class Return100Nested_Mutant5 {
    //@ ensures \result == 100;
    public static int return100 () {
        int res = 0;
        //@ maintaining 0 <= j && j <= 10;
        //@ maintaining res == 10 * j;
        for(int j = 0; j < 10; j++) {
            //@ maintaining 0 <= i && i <= 10;
            //@ maintaining res == 10 * j + i;
            for(int i = 0; i < 10; i++) {
                res = res + 1;
            }
        }
        return res;
    }
}
