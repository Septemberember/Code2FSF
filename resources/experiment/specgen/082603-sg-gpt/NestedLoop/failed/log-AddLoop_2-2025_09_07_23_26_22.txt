public class AddLoop_Mutant2 {
    //@ spec_public
    public static int AddLoop(int x, int y) {
        int sum = x;
        if (y > 0) {
            int n = y;
            //@ maintaining 0 <= n && n <= y;
            //@ maintaining sum == x + (y - n);
            //@ decreases n;
            while (n > 0) {
                sum = sum + 1;
                n = n - 1;
            }
        } else {
            int n = -y;
            //@ maintaining 0 <= n && n <= -y;
            //@ maintaining sum == x - 2 * (-y - n);
            //@ decreases n;
            while (n > 0) {
                sum = sum - 2;
                n = n - 1;
            }
        }
        return sum;
    }
}
