public class WaterFee_Mutant3 {
    //@ ensures \result >= 0;
    //@ ensures (\forall int i; 0 <= i && i < Integer.MAX_VALUE; \result >= calcWaterFee(i));
    public static int calcWaterFee(int tons) {
        int n = tons;
        int fee = 0;
        //@ maintaining fee >= 0 && 0 <= n;
        //@ decreases n;
        while (n > 0) {
            if (n > 10) {
                fee += 5;
            } else if (n > 3) {
                fee += 4;
            } else {
                fee += 1;
            }
            n--;
        }
        return fee;
    }
}
