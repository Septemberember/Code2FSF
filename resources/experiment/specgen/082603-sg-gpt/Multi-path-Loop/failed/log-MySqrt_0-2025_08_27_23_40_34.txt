public class MySqrt_Original {
    //@ ensures \result >= 0 && (\result * \result <= x) && ((\result + 1) * (\result + 1) > x);
    public static int mySqrt(int x) {
        int l = 0, r = x, ans = -1;
	//@ maintaining 0 <= l && l <= r;
	//@ maintaining 0 <= r && r <= x;
	//@ maintaining -1 <= ans && ans <= x;
	//@ maintaining (\forall int i; 0 <= i && i < l; i * i <= x);
        //@ maintaining (\forall int i; r < i && i <= x; i * i > x);
        while (l <= r) {
            int mid = l + (r - l) / 2;
            if ((long) mid * mid <= x) {
                ans = mid;
                l = mid + 1;
            } else {
                r = mid - 1;
            }
        }
        return ans;
    }
}
