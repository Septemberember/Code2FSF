    //@ ensures \result == x - (\sum int i; 0 < i && i <= x; (i % 2 == 0) ? 2 : 0) <==> x + 1 >= 0;  
    public static int addHalfSelf(int x){
        int n = x;
        int addNum = 0;
        if(x < 0){
            n = -x;
        }
        //@ maintaining n >= 0;
 
        //@ decreases n;
        while(n > 0){
            if(n % 2 == 0){
                addNum += 2;
            }
            n--;
        }
        if(x < 0){
            return x - addNum;
        }
        else{
            return x + addNum;
        }
    }
