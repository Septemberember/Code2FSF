public class Return100_Original {
    //@ ensures \result == 100;
    public static int return100 () {
        int res = 0;
        int i = 0;
        //@ maintaining i >= 0 && i <= 100;
        //@ maintaining res == i;
        //@ decreases 100 - i;
        for(i = 0; i < 100; i++) {
            res = res + 1;
        }
        return res;
    }
}
