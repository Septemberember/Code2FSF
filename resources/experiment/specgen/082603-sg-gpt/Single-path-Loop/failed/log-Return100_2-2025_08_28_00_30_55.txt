public class Return100_Mutant2 {
    //@ ensures \result == 100;
    public static int return100 () {
        int res = 0;
        int i = 0;
        //@ maintaining 0 <= i && i <= 100;
        //@ maintaining res == i;
        //@ decreases 100 - i;
        for(i = 0; i <= 100; i++) { // CHANGED: condition changed to <=
            res = res + 1;
        }
        return res;
    }
}
