public class EchoIntLoop_Mutant1 {
    //@ ensures \result == x + 1;
    public static int echo(int x) {
        int res = 0;
        //@ maintaining 0 <= i && i <= x + 1;
        //@ maintaining res == i;
        //@ decreases x - i;
        for(int i = 0; i <= x; i++) {
            res = res + 1;
        }
        return res;
    }
}
