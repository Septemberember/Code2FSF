public class Return100_Mutant1 {
    //@ ensures \result == 100;
    public static int return100() {
        int res = 0;
        int i = 0;
        //@ maintaining 0 <= i && i <= 100;
        //@ maintaining res == 2 * i;
        for (i = 0; i < 100; i++) {
            res = res + 2; // CHANGED: incremented by 2 instead of 1
        }
        return res;
    }
}
