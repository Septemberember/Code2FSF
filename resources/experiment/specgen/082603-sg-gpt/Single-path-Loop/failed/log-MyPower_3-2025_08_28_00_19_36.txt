public class MyPower_Mutant3 {
    //@ requires Integer.MIN_VALUE <= x && x <= Integer.MAX_VALUE && Integer.MIN_VALUE <= n && n <= Integer.MAX_VALUE;
    //@ ensures \result == (int)Math.pow(x, n);
    public static int power(int x, int n) {
        int res = 1;
        //@ maintaining res == (int)Math.pow(x, i) && 0 <= i && i <= n;
        //@ decreasing n - i;
        for(int i = 0; i < n; i++){
	        res = res + x;
        }
        return res;
    }
}
