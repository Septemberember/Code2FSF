public class MyPower_Mutant4 {
    //@ ensures \result == x^n;
    public static int power(int x, int n) {
        int res = 1;
        //@ maintaining 0 <= i && i <= n + 1;
        //@ maintaining res == x^i;
        //@ decreases n - i;
        for(int i = 0; i <= n; i++){
            res = res * x;
        }
        return res;
    }
}
