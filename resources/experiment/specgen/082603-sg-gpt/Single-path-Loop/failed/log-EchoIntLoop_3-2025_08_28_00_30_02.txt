public class EchoIntLoop_Mutant3 {
    //@ ensures \result == 0;
    public static int echo(int x) {
        int res = 0;
        //@ maintaining 1 <= i && i <= x;
        //@ maintaining res == i - 1;
        //@ decreases x - i;
        for(int i = 1; i < x; i++) {
            res = res + 1;
        }
        return res;
    }
}
