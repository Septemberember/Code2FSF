public class MyPower_Mutant2 {
    //@ ensures \result == (\product int i; 1 <= i && i <= n; x);
    public static int power(int x, int n) {
        int res = 1;
        //@ maintaining 1 <= i && i <= n;
        //@ maintaining res == (\product int j; 1 <= j && j < i; x);
        //@ decreases n - i;
        for(int i = 1; i <= n; i++){
            res = res * x;
        }
        return res;
    }
}
