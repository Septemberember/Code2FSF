public class SmallestEvenMulBranch_Mutant2 {
    //@ requires n >= Integer.MIN_VALUE && n <= Integer.MAX_VALUE;
    //@ ensures n % 2 == 0 ==> \result == n + 1;
    //@ ensures !(n % 2 == 0) ==> \result == 2 * n;
    public static int smallestEvenMultiple(int n) {
        if (n % 2 == 0) {
            return n + 1;
        }
        return 2 * n;
    }
}
