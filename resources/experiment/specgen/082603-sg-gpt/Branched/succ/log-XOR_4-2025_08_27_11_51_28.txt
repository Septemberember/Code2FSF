public class XOR_Mutant4 {
    //@ requires true;
    //@ ensures \result == (b1 ^ b2);
    /*@ pure @*/ public static boolean xor(boolean b1, boolean b2) {
        if(b1) {
            if(b2) {
                return false;
            }
            else {
                return true;
            }
        }
        else {
            if(b2) {
                return true;
            }
            else {
                return false;
            }
        }
    }
}
