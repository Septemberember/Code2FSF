public class ChangeCase_Original {

    //@ ensures \result == c ==> (c > 'z' || (c >= 'a' && \result == (char)(c - 'a' + 'A')) || c > 'Z' || (c >= 'A' && \result == (char)(c - 'A' + 'a')) || \result == c);
    public static char changeCase(char c) {
        char result = ' ';
        if (c > 'z') {
            result = c;
        } else if (c >= 'a') {
            result = (char) (c - 'a' + 'A');
        } else if (c > 'Z') {
            result = c;
        } else if (c >= 'A') {
            result = (char) (c - 'A' + 'a');
        } else {
            result = c;
        }
        return result;
    }
}
