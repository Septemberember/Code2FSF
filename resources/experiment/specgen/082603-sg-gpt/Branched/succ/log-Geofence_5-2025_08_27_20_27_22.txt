public class Geofence_Mutant5 {
    //@ requires Integer.MIN_VALUE < x && x < Integer.MAX_VALUE;
    //@ requires Integer.MIN_VALUE < y && y < Integer.MAX_VALUE;
    //@ requires Integer.MIN_VALUE < limitX && limitX < Integer.MAX_VALUE;
    //@ requires Integer.MIN_VALUE < limitY && limitY < Integer.MAX_VALUE;
    //@ ensures \result == 1 || \result == 2;
    public static int geofence(int x, int y, int limitX, int limitY) {
        int alarm = 2;
        if (x < 0 || x > limitX) {
            alarm = 1;
        }
        if (y < 0 || y > limitY) {
            alarm = 1;
        }
        return alarm;
    }
}
