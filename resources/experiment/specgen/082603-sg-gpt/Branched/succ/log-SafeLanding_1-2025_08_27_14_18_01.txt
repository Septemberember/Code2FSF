public class SafeLanding_Mutant1 {
    //@ ensures height <= 5 ==> \result == 0;
    //@ ensures speed > 3 ==> \result == 0;
    //@ ensures tilt > 10 ==> \result == 0;
    //@ ensures height > 5 && speed <= 3 && tilt <= 10 ==> \result == 1;
    public static int safeLanding(int height, int speed, int tilt) {
        int safe = 1;
        if (height <= 5) {
            safe = 0;
        }
        if (speed > 3) {
            safe = 0;
        }
        if (tilt > 10) {
            safe = 0;
        }
        return safe;
    }
}
