public class SmallestEvenMulBranch_Mutant4 {
    //@ requires n != Integer.MIN_VALUE;
    //@ ensures n % 2 == 0 ==> \result == n;
    //@ ensures n % 2 != 0 ==> \result == 3 * n;
    public static int smallestEvenMultiple(int n) {
        if (n % 2 == 0) {
            return n;
        }
        return 3 * n;
    }
}
