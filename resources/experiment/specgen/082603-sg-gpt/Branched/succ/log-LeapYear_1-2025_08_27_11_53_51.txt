public class LeapYear_Mutant1 {
    //@ ensures \result == true <==> (year % 4 == 0 && (year % 100 != 0 || year % 400 == 0));
    //@ ensures \result == false <==> !(year % 4 == 0 && (year % 100 != 0 || year % 400 == 0));
    public static boolean isLeapYear(int year) {
        boolean leap = true;
        if (year % 4 == 0) {
            if (year % 100 == 0) {
                if (year % 400 == 0)
                    leap = true;
                else
                    leap = false;
            } else
                leap = true;
        } else
            leap = false;
        return leap;
    }
}
