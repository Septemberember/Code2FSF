public class Conjunction_Mutant1 {
    //@ ensures \result <==> (b1 == false) && (b2 == true);
    public static boolean conjunctOf(boolean b1, boolean b2) {
        if(b1 == true)
            return false;
        if(b2 == false)
            return false;
        return true;
    }
}
