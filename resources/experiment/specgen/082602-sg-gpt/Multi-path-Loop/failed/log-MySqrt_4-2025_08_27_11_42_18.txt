public class MySqrt_Mutant4 {
    //@ requires x >= 0;
    //@ ensures \result * \result <= x && x < (\result + 1) * (\result + 1);
    public static int mySqrt(int x) {
        int l = 0, r = x, ans = -1;
        //@ maintaining l <= r;
        //@ maintaining (\forall int i; 0 <= i && i < l; (long)i * i < x);
        //@ maintaining (\forall int i; r < i && i <= x; x <= (long)i * i);
        while (l <= r) {
            int mid = l + (r - l) / 2;
            if ((long) mid * mid < x) {
                ans = mid;
                l = mid + 1;
            } else {
                r = mid - 1;
            }
        }
        return ans;
    }
}
