public class WaterFee_Mutant3 {
    //@ requires tons >= 0;
    //@ ensures \result >= 0;
    public static int calcWaterFee(int tons) {
        int n = tons;
        int fee = 0;
        //@ maintaining n >= 0;
        //@ maintaining fee >= 0;
        //@ decreasing n;
        while (n > 0) {
            if (n > 10) {
                fee += 5;
            } else if (n > 3) {
                fee += 4;
            } else {
                fee += 1;
            }
            n--;
        }
        return fee;
    }
}
