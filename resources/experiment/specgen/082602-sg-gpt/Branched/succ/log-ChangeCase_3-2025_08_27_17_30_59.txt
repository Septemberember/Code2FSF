public class ChangeCase_Mutant3 {

    //@ ensures \result >= 'A' && \result <= 'Z' <==> c >= 'a' && c <= 'z';
    //@ ensures \result >= 'a' && \result <= 'z' <==> c >= 'A' && c <= 'Z';
    //@ ensures \result == c <==> !(c >= 'A' && c <= 'Z') && !(c >= 'a' && c <= 'z');
    public static char changeCase(char c) {
        char result = ' ';
        if (c > 'z') {
            result = c;
        } else if (c >= 'a') {
            result = (char) (c - 'a' + 'A');
        } else if (c > 'Z') {
            result = c;
        } else if (c >= 'A') {
            result = (char) (c - 'A' + 'a');
        } else {
            result = c;
        }
        return result;
    }
}
