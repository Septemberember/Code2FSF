public class NegAbs_Original {

    /*@ public normal_behavior
      @    requires num >= Integer.MIN_VALUE && num <= Integer.MAX_VALUE;
      @    ensures num < 0 ==> \result == num && num >= 0 ==> \result == -num; @*/
    public static int negAbs(int num) {
        if (num < 0)
            return num;
        else
            return -num;
    }
}
