public class Disjunction_Mutant1 {
    //@ ensures \result <==> (b1 == false) || (b2 == true);
    //@ ensures !\result <==> !((b1 == false) || (b2 == true));
    public static boolean disjunctOf(boolean b1, boolean b2) {
        if(b1 == false)
            return true;
        if(b2 == true)
            return true;
        return false;
    }
}
