public class NegAbs_Mutant4 {
    //@ requires Integer.MIN_VALUE <= num && num <= Integer.MAX_VALUE;
    //@ ensures \result >= 0 && (\result == num || \result == -num);
    public static int negAbs(int num) {
        if (num > 0)
            return num;
        else
            return -num;
    }
}
