public class ReLU_Mutant3 {
    //@ ensures x < 0.0 ==> \result == x;
    //@ ensures !(x < 0.0) ==> \result == 0.0;
    public static /*@ pure @*/ double computeReLU(double x) {
        if(x < 0.0) {
            return x;
        }
        return 0.0;
    }
}
