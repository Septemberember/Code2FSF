public class AbsSeq_Mutant1 {
    //@ requires num >= Integer.MIN_VALUE && num <= Integer.MAX_VALUE;
    //@ ensures \result >= 0 && \result <= Integer.MAX_VALUE;
    //@ ensures num >= 0 ==> \result == num;
    //@ ensures num < 0 ==> \result == -num;
    public static int Abs(int num) {
        if (num <= 0) {
            num = -num;
            return num;
        } else {
            return num;
        }
    }
}
